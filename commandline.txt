Ordering
mysql> select * from Authors order by lastName; //will order by lastName
mysql> select * from Authors order by firstName, lastName; //will order by firstName and lastName
mysql> select * from Authors order by firstName desc; //will order in descending order
mysql> select * from Authors order by firstName asc, lastName desc;
